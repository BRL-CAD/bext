
bext_enable(APPLESEED)

if (ENABLE_APPLESEED)

  git_submodule_init(appleseed CMakeLists.txt)

  RegisterDeps(APPLESEED)

  ExternalProject_Add(APPLESEED_BLD
    SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/appleseed"
    BUILD_ALWAYS ${EXT_BUILD_ALWAYS} ${LOG_OPTS}
    #PATCH_COMMAND ${PATCH_EXECUTABLE};-E;-p1;${PATCH_OPTIONS};-i;${CMAKE_CURRENT_SOURCE_DIR}/appleseed.patch
    CMAKE_ARGS
    -DCMAKE_FIND_PACKAGE_PREFER_CONFIG=ON
    ${BUILD_TYPE_SPECIFIER}
    $<$<BOOL:${BOOST_TARGET}>:-DBOOST_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${EIGEN_BUNDLED}>:-DEigen3_ROOT=${CMAKE_NOBUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${EMBREE_TARGET}>:-DEMBREE_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${ISPC_TARGET}>:-DISPC_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${JPEG_TARGET}>:-DJPEG_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${LLVM_TARGET}>:-DLLVM_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${LZ4_TARGET}>:-DLZ4_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${OPENCOLORIO_TARGET}>:-DOPENCOLORIO_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${OPENEXR_TARGET}>:-DOPENEXR_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${OPENIMAGEIO_TARGET}>:-DOPENIMAGEIO_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${OSL_TARGET}>:-DOSL_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${PNG_TARGET}>:-DPNG_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${TBB_TARGET}>:-DTBB_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${TIFF_TARGET}>:-DTIFF_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${XERCES_C_TARGET}>:-DXERCES_C_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    $<$<BOOL:${ZLIB_TARGET}>:-DZLIB_ROOT=${CMAKE_BUNDLE_INSTALL_PREFIX}>
    -DWITH_STUDIO=OFF
    -DWITH_BENCH=OFF
    -DWITH_TOOLS=OFF
    -DWITH_PYTHON2_BINDINGS=OFF
    -DWITH_PYTHON3_BINDINGS=OFF
    -DINSTALL_TESTS=OFF
    -DINSTALL_API_EXAMPLES=OFF
    -DUSE_FIND_PACKAGE_FOR_EMBREE=ON
    -DUSE_FIND_PACKAGE_FOR_EXR=ON
    -DUSE_FIND_PACKAGE_FOR_LZ4=ON
    -DUSE_FIND_PACKAGE_FOR_OCIO=ON
    -DUSE_FIND_PACKAGE_FOR_OIIO=ON
    -DUSE_FIND_PACKAGE_FOR_OSL=ON
    -DUSE_FIND_PACKAGE_FOR_XERCES=ON
    -DUSE_FIND_PACKAGE_FOR_ZLIB=ON
    -DBIN_DIR=${BIN_DIR}
    -DLIB_DIR=${LIB_DIR}
    -DBUILD_STATIC_LIBS=${BUILD_STATIC_LIBS}
    -DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER}
    -DCMAKE_C_COMPILER=${CMAKE_C_COMPILER}
    -DCMAKE_INSTALL_PREFIX=${CMAKE_BUNDLE_INSTALL_PREFIX}
    -DCMAKE_INSTALL_LIBDIR:PATH=${LIB_DIR}
    -DCMAKE_INSTALL_RPATH=${CMAKE_BUNDLE_INSTALL_PREFIX}/${LIB_DIR}
    LOG_CONFIGURE ${EXT_BUILD_QUIET}
    LOG_BUILD ${EXT_BUILD_QUIET}
    LOG_INSTALL ${EXT_BUILD_QUIET}
    LOG_OUTPUT_ON_FAILURE ${EXT_BUILD_QUIET}
    STEP_TARGETS install
    )

  TargetInstallDeps(APPLESEED APPLESEED_DEPENDS)

  # Copy the license into position in CMAKE_BUNDLE_INSTALL_PREFIX
  configure_file(
    ${CMAKE_CURRENT_SOURCE_DIR}/appleseed/LICENSE.txt
    ${DOC_LICENSE_DIR}/appleseed.txt
    COPYONLY
    )

endif (ENABLE_APPLESEED)
